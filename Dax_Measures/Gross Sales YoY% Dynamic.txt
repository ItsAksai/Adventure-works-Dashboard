Gross Sales YoY% Dynamic = 
VAR _LatestVisiblePeriod =
    MAX ( Sales[OrderDate] )

VAR _CurrentMonthStart =
    DATE ( YEAR ( _LatestVisiblePeriod ), MONTH ( _LatestVisiblePeriod ), 1 )

VAR _Previous12MonthSales =
    CALCULATE (
        SUM ( Sales[Sales] ),
        REMOVEFILTERS ( Sales ),
        DATESBETWEEN (
            Sales[OrderDate],
            EOMONTH ( _CurrentMonthStart, -24 ) + 1,
            EOMONTH ( _CurrentMonthStart, -12 )
        )
    )

VAR _Current12MonthSales =
    CALCULATE (
        SUM ( Sales[Sales] ),
        REMOVEFILTERS ( Sales ),
        DATESBETWEEN (
            Sales[OrderDate],
            EOMONTH ( _CurrentMonthStart, -12 ) + 1,
            EOMONTH ( _CurrentMonthStart, 0 )
        )
    )

VAR _Result =
    IF (
        _Previous12MonthSales > 0,
        (_Current12MonthSales - _Previous12MonthSales) / _Previous12MonthSales,
        BLANK ()
    )

RETURN
    _Result
